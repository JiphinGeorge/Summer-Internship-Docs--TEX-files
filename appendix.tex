\section*{Appendix F – Source Code Snippets}
\textbf{QualiVision AI - Flask Prediction Endpoint (app.py):}
\begin{verbatim}
@app.route('/api/predict', methods=['POST'])
def predict():
    if 'image' not in request.files:
        return jsonify({'error': 'No image uploaded'}), 400
    
    file = request.files['image']
    image_bytes = file.read()
    
    # Preprocess image
    processed_image = preprocess_image(image_bytes)
    
    # Perform inference
    predictions = model.predict(processed_image)
    confidence = np.max(predictions)
    predicted_class = class_names[np.argmax(predictions)]
    
    # Generate Grad-CAM
    heatmap_path = generate_gradcam(processed_image, predicted_class)
    
    # Log to database
    log_prediction_to_db(predicted_class, confidence, heatmap_path)
    
    return jsonify({
        'status': 'success',
        'prediction': predicted_class,
        'confidence': float(confidence),
        'gradcam_url': heatmap_path
    })
\end{verbatim}
